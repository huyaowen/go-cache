### 步骤 3: 定义 Service 接口和实现

```go
// service/user.go
package service

//go:generate gocache scan .

import (
    "database/sql"
    "your-project/model"
)

// UserServiceInterface 定义服务接口
type UserServiceInterface interface {
    GetUser(id int64) (*model.User, error)
    CreateUser(name, email string) (*model.User, error)
    UpdateUser(id int64, name, email string) (*model.User, error)
    DeleteUser(id int64) error
}

// userService 服务实现
type userService struct {
    db *sql.DB
}

// NewUserServiceRaw 创建原始服务（不带缓存）
func NewUserServiceRaw(db *sql.DB) *userService {
    return &userService{db: db}
}

// GetUser - 带缓存读取
// @cacheable(cache="users", key="#id", ttl="30m")
func (s *userService) GetUser(id int64) (*model.User, error) {
    // 业务逻辑：查询数据库
    var user model.User
    err := s.db.QueryRow("SELECT * FROM users WHERE id = ?", id).Scan(&user.ID, &user.Name, &user.Email)
    return &user, err
}

// CreateUser - 带缓存更新
// @cacheput(cache="users", key="#result.ID", ttl="30m")
func (s *userService) CreateUser(name, email string) (*model.User, error) {
    // 业务逻辑：插入数据
    result, err := s.db.Exec("INSERT INTO users (name, email) VALUES (?, ?)", name, email)
    if err != nil {
        return nil, err
    }
    id, _ := result.LastInsertId()
    return &model.User{ID: id, Name: name, Email: email}, nil
}

// UpdateUser - 带缓存更新
// @cacheput(cache="users", key="#id", ttl="30m")
func (s *userService) UpdateUser(id int64, name, email string) (*model.User, error) {
    // 业务逻辑：更新数据
    _, err := s.db.Exec("UPDATE users SET name = ?, email = ? WHERE id = ?", name, email, id)
    if err != nil {
        return nil, err
    }
    return &model.User{ID: id, Name: name, Email: email}, nil
}

// DeleteUser - 带缓存清除
// @cacheevict(cache="users", key="#id")
func (s *userService) DeleteUser(id int64) error {
    // 业务逻辑：删除数据
    _, err := s.db.Exec("DELETE FROM users WHERE id = ?", id)
    return err
}
```

**生成代码示例** (`service/user_cached.go`):

```go
// Code generated by go-cache-gen. DO NOT EDIT.

package service

import (
    "context"
    "time"
    
    "github.com/coderiser/go-cache/pkg/cache"
    "github.com/coderiser/go-cache/pkg/core"
    "github.com/coderiser/go-cache/pkg/spel"
)

// cachedUserService 缓存包装器
type cachedUserService struct {
    raw       *userService
    manager   core.CacheManager
    evaluator *spel.SpELEvaluator
}

// UserService 全局实例
var UserService UserServiceInterface

// NewCachedUserService 创建带缓存的服务实例
func NewCachedUserService(db *sql.DB) UserServiceInterface {
    raw := NewUserService(db)
    return &cachedUserService{
        raw:       raw,
        manager:   cache.GetGlobalManager(),
        evaluator: spel.NewSpELEvaluator(),
    }
}

// InitUserService 初始化全局实例
func InitUserService(db *sql.DB) {
    UserService = NewCachedUserService(db)
}

func (c *cachedUserService) GetUser(id int64) (*model.User, error) {
    ctx := context.Background()
    cache, _ := c.manager.GetCache("users")
    
    // 构建缓存 Key
    evalCtx := spel.NewEvaluationContext()
    evalCtx.SetArg("id", id)
    evalCtx.SetArgByIndex(0, id)
    key, _ := c.evaluator.EvaluateToString("#id", evalCtx)
    
    // 查缓存
    if val, found, _ := cache.Get(ctx, key); found {
        return val.(*model.User), nil
    }
    
    // 调用原始方法
    result, err := c.raw.GetUser(id)
    if err != nil {
        return nil, err
    }
    
    // 写缓存
    cache.Set(ctx, key, result, 30*time.Minute)
    
    return result, nil
}

// ... 其他方法类似
```

### 步骤 5: 配置缓存管理器

```go
// main.go
package main

import (
    "log"
    "time"
    
    "github.com/coderiser/go-cache/pkg/cache"
    "github.com/coderiser/go-cache/pkg/core"
    "github.com/coderiser/go-cache/pkg/backend"
    "your-project/service"
)

func main() {
    // 1. 创建缓存管理器
    manager := core.NewCacheManager()
    
    // 2. 配置 Redis 后端（生产环境推荐）
    redisBackend, err := backend.NewRedisBackend(&backend.RedisConfig{
        Addr:       "localhost:6379",
        Password:   "",  // 如有密码则填写
        DB:         0,
        Prefix:     "myapp",
        DefaultTTL: 30 * time.Minute,
        PoolSize:   10,
    })
    if err != nil {
        log.Printf("Redis 连接失败，降级到内存后端：%v", err)
        
        // 降级到 Memory 后端（开发环境可用）
        redisBackend = backend.NewMemoryBackend(&backend.CacheConfig{
            Name:           "users",
            MaxSize:        10000,
            DefaultTTL:     30 * time.Minute,
            EvictionPolicy: "lru",
        })
    }
    
    // 3. 注册缓存
    manager.RegisterCache("users", redisBackend)
    
    // 4. 设置为全局管理器
    cache.SetGlobalManager(manager)
    
    // 5. 优雅关闭（在程序退出时调用）
    defer cache.CloseGlobalManager()
    
    // 6. 初始化数据库
    db := initDB()
    
    // 7. 服务会在使用时自动创建（无需手动初始化）
    // 直接使用：userService := service.NewUserService(db)
    
    // 8. 运行业务逻辑
    runServer(db)
}
```

## 🎯 预期效果

### 功能效果

| 注解 | 行为 | 预期效果 |
|------|------|----------|
| `@cacheable` | 方法执行前检查缓存 | 命中则直接返回，未命中则执行方法 + 写入缓存 |
| `@cacheput` | 总是执行方法 | 执行后将结果写入缓存 |
| `@cacheevict` | 方法执行后清除缓存 | 删除指定 Key 的缓存 |

### 性能效果

| 场景 | 预期延迟 | 说明 |
|------|----------|------|
| Memory 命中 | < 1ms | 纯内存操作 |
| Redis 命中 | < 5ms | 网络 + 序列化 |
| 缓存未命中 | 数据库查询时间 | 首次查询或过期 |
| SpEL 求值 | < 50μs | expr 引擎 |

### 代码效果

**Before (无缓存)**:
```go
func GetUser(id int64) (*User, error) {
    // 每次调用都查询数据库
    var user User
    err := db.QueryRow("SELECT * FROM users WHERE id = ?", id).Scan(&user.ID, &user.Name, &user.Email)
    return &user, err
}
```

**After (带缓存)**:
```go
// @cacheable(cache="users", key="#id", ttl="30m")
func GetUser(id int64) (*User, error) {
    // 缓存命中时直接返回
    // 未命中时查询数据库 + 写入缓存
    var user User
    err := db.QueryRow("SELECT * FROM users WHERE id = ?", id).Scan(&user.ID, &user.Name, &user.Email)
    return &user, err
}
```

**使用方代码不变**:
```go
user, err := service.UserService.GetUser(1)  // 缓存自动生效
```

## ✅ 验证方法

### 1. 基础功能验证

```go
// tests/cache_test.go
package tests

import (
    "testing"
    "your-project/service"
    "github.com/stretchr/testify/assert"
)

func TestCacheable(t *testing.T) {
    // 创建服务（带缓存）
    db := initTestDB()
    userService := service.NewUserService(db)
    
    // 第一次调用 - 缓存未命中
    user1, err := userService.GetUser(1)
    assert.NoError(t, err)
    assert.NotNil(t, user1)
    
    // 第二次调用 - 缓存命中
    user2, err := userService.GetUser(1)
    assert.NoError(t, err)
    assert.Equal(t, user1.ID, user2.ID)
}

func TestCacheput(t *testing.T) {
    db := initTestDB()
    userService := service.NewUserService(db)
    
    // 创建用户 - 写入缓存
    user, err := userService.CreateUser("Test", "test@example.com")
    assert.NoError(t, err)
    assert.NotNil(t, user)
    
    // 获取用户 - 应该命中缓存
    cachedUser, err := userService.GetUser(user.ID)
    assert.NoError(t, err)
    assert.Equal(t, user.ID, cachedUser.ID)
}

func TestCacheevict(t *testing.T) {
    db := initTestDB()
    userService := service.NewUserService(db)
    
    // 先创建用户
    user, _ := userService.CreateUser("Test", "test@example.com")
    
    // 删除用户 - 清除缓存
    err := userService.DeleteUser(user.ID)
    assert.NoError(t, err)
    
    // 再次获取 - 应该查询数据库（缓存已清除）
    _, err = userService.GetUser(user.ID)
    assert.Error(t, err)  // 用户不存在
}
```

## 🔍 常见问题排查

### Q1: 生成的代码在哪里？

**A**: 在扫描目录中，与源文件同目录，文件名格式为 `xxx_cached.go`。

例如：`service/user.go` → `service/user_cached.go`

### Q2: 注解不生效？

**排查步骤**:
1. 检查注解格式是否正确：`// @cacheable(cache="users", key="#id", ttl="30m")`
2. 检查是否执行了 `gocache scan ./service`
3. 检查生成的 `*_cached.go` 文件是否存在
4. 检查是否导入了生成的包

### Q3: 如何切换缓存后端？

**A**: 在 `main()` 中配置自定义 Manager：

```go
manager := core.NewCacheManager()
redisBackend := backend.NewRedisBackend(...)
manager.RegisterCache("users", redisBackend)
cache.SetGlobalManager(manager)
```

### Q4: 如何监控缓存命中率？

**A**: 使用 `cache.Stats()` 或启用 Prometheus 指标：

```go
manager.EnableMetrics()
// 在 /metrics 端点暴露
```

### Q5: 如何处理缓存穿透？

**A**: 框架内置空值缓存保护，也可手动配置：

```go
config := &core.ProtectionConfig{
    EnablePenetrationProtection: true,
    EmptyValueTTL:               5 * time.Minute,
}
```

### Q6: SpEL 表达式支持哪些变量？

**A**: 
- `#id`, `#name` 等：方法参数
- `#result`: 方法返回值（仅 `@cacheput` 可用）
- `#arg0`, `#arg1` 等：按索引访问参数

示例：
```go
// @cacheable(cache="users", key="#id", ttl="30m")
func GetUser(id int64) (*User, error)

// @cacheput(cache="users", key="#result.ID", ttl="30m")
func CreateUser(name string) (*User, error)
```

### Q7: 如何处理错误？

**A**: 生成的代码会自动处理错误，缓存失败时降级调用原始方法：

```go
cache, err := c.manager.GetCache("users")
if err != nil {
    // 降级：直接调用原始方法
    return c.raw.GetUser(id)
}
```
